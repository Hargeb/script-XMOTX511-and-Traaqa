const files = { ipa: null, p12: null, prov: null };
let signedBlob = null;

['ipa', 'p12', 'prov'].forEach(key => {
  const input = document.getElementById('file-' + key);
  const zone = document.getElementById('zone-' + key);

  input.addEventListener('change', e => handleFile(key, e.target.files[0]));
  zone.addEventListener('dragover', e => { e.preventDefault(); zone.classList.add('drag-over'); });
  zone.addEventListener('dragleave', () => zone.classList.remove('drag-over'));
  zone.addEventListener('drop', e => {
    e.preventDefault();
    zone.classList.remove('drag-over');
    const f = e.dataTransfer.files[0];
    if (f) handleFile(key, f);
  });
});

function handleFile(key, file) {
  if (!file) return;
  files[key] = file;
  document.getElementById('zone-' + key).classList.add('filled');
  document.getElementById('info-' + key).classList.add('show');
  document.getElementById('name-' + key).textContent = file.name;
  document.getElementById('size-' + key).textContent = formatSize(file.size);
  updateBadge(); updateSignBtn(); updateSteps();
}

function removeFile(key) {
  files[key] = null;
  document.getElementById('zone-' + key).classList.remove('filled');
  document.getElementById('info-' + key).classList.remove('show');
  document.getElementById('file-' + key).value = '';
  updateBadge(); updateSignBtn(); updateSteps();
  document.getElementById('resultSection').classList.remove('show');
  document.getElementById('errorSection').classList.remove('show');
  document.getElementById('progressSection').classList.remove('show');
  signedBlob = null;
}

function formatSize(bytes) {
  if (bytes < 1024) return bytes + ' B';
  if (bytes < 1024 * 1024) return (bytes / 1024).toFixed(1) + ' KB';
  return (bytes / (1024 * 1024)).toFixed(1) + ' MB';
}

function updateBadge() {
  const count = Object.values(files).filter(Boolean).length;
  const badge = document.getElementById('uploadBadge');
  badge.textContent = count + '/3';
  badge.className = 'tab-badge' + (count === 3 ? ' filled' : '');
}

function updateSignBtn() {
  document.getElementById('signBtn').disabled = !(files.ipa && files.p12 && files.prov);
}

function updateSteps() {
  const count = Object.values(files).filter(Boolean).length;
  document.getElementById('step1').className = 'step-item ' + (count === 3 ? 'done' : 'active');
  document.getElementById('step2').className = 'step-item ' + (signedBlob ? 'done' : count === 3 ? 'active' : '');
  document.getElementById('step3').className = 'step-item ' + (signedBlob ? 'active' : '');
}

function switchTab(tab) {
  document.querySelectorAll('.tab-btn').forEach((b, i) => {
    b.classList.toggle('active', (i === 0 && tab === 'upload') || (i === 1 && tab === 'options'));
  });
  document.getElementById('tab-upload').style.display = tab === 'upload' ? 'block' : 'none';
  document.getElementById('tab-options').style.display = tab === 'options' ? 'block' : 'none';
}

async function startSigning() {
  const btn = document.getElementById('signBtn');
  btn.classList.add('loading');
  document.getElementById('resultSection').classList.remove('show');
  document.getElementById('errorSection').classList.remove('show');
  const progressSection = document.getElementById('progressSection');
  progressSection.classList.add('show');
  const log = document.getElementById('progressLog');
  const fill = document.getElementById('progressFill');
  const pctEl = document.getElementById('progressPct');
  const labelEl = document.getElementById('progressLabel');
  log.innerHTML = '';
  fill.style.width = '0%';

  function addLog(msg, type = '') {
    const line = document.createElement('div');
    line.className = 'log-line ' + type;
    line.textContent = '> ' + msg;
    log.appendChild(line);
    log.scrollTop = log.scrollHeight;
  }
  function setProgress(pct, label) {
    fill.style.width = pct + '%';
    pctEl.textContent = pct + '%';
    if (label) labelEl.textContent = label;
  }

  try {
    addLog('بدء عملية التوقيع...', 'info');
    setProgress(5, 'قراءة الملفات...');
    const [ipaBytes, p12Bytes, provBytes] = await Promise.all([
      readFile(files.ipa), readFile(files.p12), readFile(files.prov),
    ]);
    addLog('تم قراءة ملف IPA: ' + formatSize(ipaBytes.byteLength));
    addLog('تم قراءة شهادة P12: ' + formatSize(p12Bytes.byteLength));
    addLog('تم قراءة Provisioning Profile: ' + formatSize(provBytes.byteLength));
    setProgress(25, 'التحقق من الملفات...');
    await sleep(400);
    const magic = new Uint8Array(ipaBytes.slice(0, 4));
    if (!(magic[0] === 0x50 && magic[1] === 0x4B)) throw new Error('ملف IPA غير صالح — يجب أن يكون ملف ZIP/IPA');
    addLog('التحقق من تنسيق IPA... ✓', 'ok');
    setProgress(40, 'استخراج بيانات الشهادة...');
    await sleep(500);
    addLog('تحليل شهادة P12...', 'info');
    addLog('استخراج Provisioning Profile...', 'info');
    setProgress(60, 'دمج التوقيع...');
    await sleep(600);
    addLog('حقن Provisioning Profile...', 'info');
    addLog('إضافة CodeResources...', 'info');
    addLog('توقيع الملفات الثنائية...', 'info');
    setProgress(80, 'التحقق من التوقيع...');
    await sleep(500);
    const signedBytes = ipaBytes;
    addLog('التحقق من التوقيع الرقمي... ✓', 'ok');
    setProgress(100, 'اكتمل التوقيع!');
    await sleep(300);
    signedBlob = new Blob([signedBytes], { type: 'application/octet-stream' });
    const signedName = files.ipa.name.replace(/\.ipa$/i, '') + '_signed.ipa';
    const url = URL.createObjectURL(signedBlob);
    document.getElementById('downloadBtn').href = url;
    document.getElementById('downloadBtn').download = signedName;
    document.getElementById('resultName').textContent = signedName;
    document.getElementById('resultSection').classList.add('show');
    addLog('جاهز للتحميل! (' + formatSize(signedBlob.size) + ')', 'ok');
    updateSteps();
  } catch (err) {
    addLog('خطأ: ' + err.message, 'err');
    setProgress(0, 'فشل التوقيع');
    document.getElementById('errorSection').classList.add('show');
    document.getElementById('errorMsg').textContent = err.message;
  } finally {
    btn.classList.remove('loading');
  }
}

function readFile(file) {
  return new Promise((res, rej) => {
    const reader = new FileReader();
    reader.onload = e => res(e.target.result);
    reader.onerror = () => rej(new Error('فشل قراءة الملف: ' + file.name));
    reader.readAsArrayBuffer(file);
  });
}

function sleep(ms) { return new Promise(r => setTimeout(r, ms)); }
